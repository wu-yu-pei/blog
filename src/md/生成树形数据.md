#### 生成树形数据

> 给出一组有父子关系的数组数据,生成对应的树形数据

##### 比如:

##### 代码实现

###### 一

```js
arr.forEach((item, index) => {
  item.children = arr.filter((citem, cindex) => {
    return citem.pid == item.id;
  });
});
let result = arr.filter((item) => item.pid == 0);
console.log(result);
```

###### 二

```js
let treeMapList = arr.reduce((pre, cur) => {
  pre[cur.id] = cur;
  return pre;
}, {});

let res = arr.reduce((pre, cur) => {
  const pid = cur.pid;
  const parent = treeMapList[pid];
  if (parent) {
    parent.children ? parent.children.push(cur) : (parent.children = [cur]);
  } else if (cur.pid == 0) {
    // 根
    pre.push(cur);
  }
  return pre;
}, []);
console.log(res);
```
