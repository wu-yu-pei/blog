  item.children = arr.filter((citem, cindex) => {
    return citem.pid == item.id;
  });

二
let treeMapList = arr.reduce((pre, cur) => {
  pre[cur.id] = cur;
  return pre;
}, {});

let res = arr.reduce((pre, cur) => {
  const pid = cur.pid;
  const parent = treeMapList[pid];
  if (parent) {
    parent.children ? parent.children.push(cur) : (parent.children = [cur]);
  } else if (cur.pid == 0) {
    // 根
    pre.push(cur);
  }
  return pre;
}, []);
console.log(res);
