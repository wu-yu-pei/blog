let ctx = $computed(() => canvasEl.getContext('2d')!);

function init() {
  ctx.strokeStyle = 'red';
  ctx.beginPath();
  ctx.moveTo(10, 10);
  ctx.lineTo(10, 60);
  ctx.closePath();
  ctx.stroke();
}

onMounted(() => {
  init();
});
