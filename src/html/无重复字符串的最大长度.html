无重复字符串的最大长度

leetcode

这个题是我 两个星期前写过的题,现在又不会了,复习一下

给你一个字符串,请找出其中无重复字符串的最大长度

li: abcdacad ---> abcd--4

var str = 'abcdacad';
let max = 0;
let j = 0;
let set = new Set();

for (let i = 0; i < str.length; i++) {
  if (!set.has(str[i])) {
    set.add(str[i]);
    max = Math.max(max, set.size);
  } else {
    while (set.has(str[i])) {
      set.delete(str[j]);
      j++;
    }
    set.add(str[i]);
  }
}
console.log(max);
