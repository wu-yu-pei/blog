// 1.this指向window
function jieliu(fn, delay) {
  // 开始时间
  let start = new Date();
  return function () {
    // 调用时的时间
    let end = new Date();
    // 如果时间只差大于delay 则 调用函数
    if (end - start >= delay) {
      fn();
      start = end;
    }
  };
}

int.oninput = jieliu(foo, 1000);
  • 修改this指向版本
function jieliu(fn, delay) {
  // 开始时间
  let start = new Date();
  return function () {
    // 调用时的时间
    let end = new Date();
    // 如果时间只差大于delay 则 调用函数
    if (end - start >= delay) {
      fn.call(this);
      start = end;
    }
  };
}

int.oninput = jieliu(foo, 1000);
